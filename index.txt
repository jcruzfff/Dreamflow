    document.addEventListener('DOMContentLoaded', function() {
        const testimonials = document.querySelectorAll('.testimonial');
        const prevArrow = document.querySelector('.prev-arrow');
        const nextArrow = document.querySelector('.next-arrow');
        let currentSlide = 0;
        let autoSlideInterval;

        function showSlide(index) {
            testimonials.forEach(testimonial => testimonial.classList.remove('active'));
            testimonials[index].classList.add('active');
            currentSlide = index;
        }

        function nextSlide() {
            const nextIndex = (currentSlide + 1) % testimonials.length;
            showSlide(nextIndex);
        }

        function prevSlide() {
            const prevIndex = (currentSlide - 1 + testimonials.length) % testimonials.length;
            showSlide(prevIndex);
        }

        // Auto-slide every 5 seconds
        function startAutoSlide() {
            autoSlideInterval = setInterval(nextSlide, 5000);
        }

        function resetAutoSlide() {
            clearInterval(autoSlideInterval);
            startAutoSlide();
        }

        // Add click handlers to arrows
        nextArrow.addEventListener('click', () => {
            nextSlide();
            resetAutoSlide();
        });

        prevArrow.addEventListener('click', () => {
            prevSlide();
            resetAutoSlide();
        });

        // Initialize first slide
        showSlide(0);
        
        // Start auto-sliding
        startAutoSlide();
    });
</script>

                    handleNavbarVisibility();
                    ticking = false;
                });
                ticking = true;
            }

<script>
    function initMobileProcessSteps() {
        const steps = document.querySelectorAll('.process-step');
        let currentStep = -1;
        let animationInterval = null;
        
        function showStep(index) {
            if (window.innerWidth > 1024) return;
            
            // Hide all steps first
            steps.forEach(step => step.classList.remove('active'));
            
            // Show the current step
            if (index >= 0 && index < steps.length) {
                steps[index].classList.add('active');
            }
        }

        function cycleSteps() {
            currentStep = (currentStep + 1) % steps.length;
            showStep(currentStep);
        }

        function startAnimation() {
            if (window.innerWidth > 1024 || animationInterval) return;
            
            // Reset state
            currentStep = -1;
            steps.forEach(step => step.classList.remove('active'));
            
            // Show first step immediately
            cycleSteps();
            
            // Set up interval for cycling through remaining steps
            animationInterval = setInterval(cycleSteps, 4000);
        }

        function stopAnimation() {
            if (animationInterval) {
                clearInterval(animationInterval);
                animationInterval = null;
            }
            steps.forEach(step => step.classList.remove('active'));
            currentStep = -1;
        }

        // Start animation when element is in viewport
        function handleVisibility(entries) {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    startAnimation();
                } else {
                    stopAnimation();
                }
            });
        }

        // Create intersection observer
        const observer = new IntersectionObserver(handleVisibility, {
            threshold: 0.3 // Start animation when 30% of the element is visible
        });

        // Observe the process steps container
        const processStepsContainer = document.querySelector('.process-steps');
        if (processStepsContainer) {
            observer.observe(processStepsContainer);
        }

        // Handle window resize
        window.addEventListener('resize', () => {
            if (window.innerWidth > 1024) {
                stopAnimation();
            } else if (document.querySelector('.process-steps').getBoundingClientRect().top < window.innerHeight) {
                startAnimation();
            }
        });

        // Initial check if element is already in viewport
        if (window.innerWidth <= 1024 && 
            document.querySelector('.process-steps').getBoundingClientRect().top < window.innerHeight) {
            startAnimation();
        }
    }

    // Initialize when DOM is ready
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', initMobileProcessSteps);
    } else {
        initMobileProcessSteps();
    }
</script>

            document.body.style.overflow = mobileMenu.classList.contains('active') ? 'hidden' : '';

        hamburgerBtn.addEventListener('click', toggleMenu);

        mobileMenuClose.addEventListener('click', toggleMenu);
        
        // Menu items click (close menu when clicking a link)
        mobileMenuItems.forEach(item => {
            item.addEventListener('click', () => {
                toggleMenu();
                // Add small delay to allow the menu to close before scrolling
                setTimeout(() => {
                    const targetId = item.getAttribute('href');
                    const targetElement = document.querySelector(targetId);
                    if (targetElement) {
                        targetElement.scrollIntoView({ behavior: 'smooth' });
                    }
                }, 300);
            });
        });
        
        // Apply now button click
        mobileApplyNow.addEventListener('click', toggleMenu);
    });
</script>

<script>
    document.addEventListener('DOMContentLoaded', function() {
        const metricCards = document.querySelectorAll('.metric-card');
        
        function handleMetricCards() {
            const isMobile = window.innerWidth < 678;
            
            metricCards.forEach(card => {
                const label = card.querySelector('.metric-label');
                const subtitle = card.querySelector('.metric-subtitle');
                
                if (isMobile) {
                    // Store original subtitle text if not already stored
                    if (!subtitle.dataset.originalText) {
                        subtitle.dataset.originalText = subtitle.textContent;
                    }
                    // Replace subtitle text with label text
                    subtitle.textContent = label.textContent;
                    // Hide the label
                    label.style.display = 'none';
                } else {
                    // Restore original subtitle text
                    if (subtitle.dataset.originalText) {
                        subtitle.textContent = subtitle.dataset.originalText;
                    }
                    // Show the label
                    label.style.display = '';
                }
            });
        }

        // Run on page load
        handleMetricCards();

        // Run on window resize with debounce
        let resizeTimeout;
        window.addEventListener('resize', function() {
            clearTimeout(resizeTimeout);
            resizeTimeout = setTimeout(handleMetricCards, 250);
        });
    });
</script>

<script>
    // Single instance of video initialization
    let videoInitialized = false;
    
    function initVideoPlayback() {
        if (videoInitialized) {
            console.log('Video already initialized');
            return;
        }

        console.log('Attempting to initialize video playback');
        const video = document.getElementById('mobileAnimation');
        
        if (!video) {
            console.error('Video element not found');
            return;
        }

        console.log('Video element found:', video);
        videoInitialized = true;

        // Track loading state
        let isLoading = true;
        let isPlaying = false;

        // Add loading event listeners
        video.addEventListener('loadstart', () => {
            console.log('Video load started');
            isLoading = true;
        });

        video.addEventListener('canplay', () => {
            console.log('Video can play');
            isLoading = false;
        });

        video.addEventListener('error', (e) => {
            const error = video.error;
            console.error('Video error:', {
                code: error ? error.code : 'unknown',
                message: error ? error.message : 'unknown',
                currentSrc: video.currentSrc
            });
        });

        // Check source support
        const sources = video.getElementsByTagName('source');
        Array.from(sources).forEach(source => {
            source.addEventListener('error', (e) => {
                console.error('Source error:', {
                    src: source.src,
                    type: source.type
                });
            });
        });

        async function playVideo() {
            if (isLoading) {
                console.log('Video still loading, waiting...');
                return;
            }

            if (!isPlaying) {
                console.log('Attempting to play video');
                try {
                    await video.play();
                    isPlaying = true;
                    console.log('Video playing successfully');
                } catch (err) {
                    console.error('Play failed:', err);
                    // Try to play muted if autoplay was blocked
                    if (!video.muted) {
                        console.log('Trying muted playback...');
                        video.muted = true;
                        try {
                            await video.play();
                            isPlaying = true;
                            console.log('Muted playback successful');
                        } catch (mutedErr) {
                            console.error('Muted playback also failed:', mutedErr);
                        }
                    }
                }
            }
        }

        function pauseVideo() {
            if (isPlaying) {
                video.pause();
                isPlaying = false;
                console.log('Video paused');
            }
        }

        // Create and configure intersection observer
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                console.log('Intersection state:', {
                    isIntersecting: entry.isIntersecting,
                    ratio: entry.intersectionRatio,
                    isLoading,
                    isPlaying
                });

                if (entry.isIntersecting && !isPlaying) {
                    playVideo();
                } else if (!entry.isIntersecting && isPlaying) {
                    pauseVideo();
                }
            });
        }, {
            threshold: [0, 0.3, 0.5, 0.7, 1]
        });

        observer.observe(video);
        console.log('Observer set up');

        // Try initial play if video is in viewport
        const rect = video.getBoundingClientRect();
        if (rect.top < window.innerHeight && rect.bottom > 0) {
            console.log('Video initially in viewport, attempting play');
            playVideo();
        }
    }

    // Initialize when DOM is ready
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', initVideoPlayback);
    } else {
        initVideoPlayback();
    }
</script>
